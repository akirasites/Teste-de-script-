local Utility = {}

function Utility:Create(className, properties)
    local instance = Instance.new(className)
    for k, v in pairs(properties) do instance[k] = v end
    return instance
end

function Utility:Tween(instance, properties, duration, style)
    duration = duration or 0.25
    style = style or Enum.EasingStyle.Quart
    local tweenInfo = TweenInfo.new(duration, style, Enum.EasingDirection.Out)
    local tween = TweenService:Create(instance, tweenInfo, properties)
    tween:Play()
    return tween
end

function Utility:RegisterTheme(instance, property, colorType)
    table.insert(Obsidian.Registry, { Instance = instance, Property = property, Type = colorType })
    instance[property] = Obsidian.Themes[Obsidian.CurrentTheme][colorType]
end

function Obsidian:SetTheme(themeName)
    if self.Themes[themeName] then
        self.CurrentTheme = themeName
        self:UpdateColors()
    end
end

function Obsidian:SetAccentColor(color)
    self.Themes[self.CurrentTheme].Accent = color
    self:UpdateColors()
end

function Obsidian:UpdateColors()
    local theme = self.Themes[self.CurrentTheme]
    for _, item in ipairs(self.Registry) do
        if item.Instance and item.Instance.Parent then
            Utility:Tween(item.Instance, { [item.Property] = theme[item.Type] }, 0.3)
        end
    end
end

                Header.MouseButton1Click:Connect(function()
                    Open = not Open
                    Utility:Tween(CpFrame, {Size = UDim2.new(1, 0, 0, Open and 130 or 35)}, 0.3, Enum.EasingStyle.Back)
                end)

                -- Hue Slider
                local HueBg = Utility:Create("Frame", { Parent = CpFrame, Size = UDim2.new(1, -20, 0, 10), Position = UDim2.new(0, 10, 0, 50), BackgroundColor3 = Color3.fromRGB(255,255,255) })
                Utility:Create("UICorner", { Parent = HueBg, CornerRadius = UDim.new(1, 0) })
                local UIGradient = Utility:Create("UIGradient", { Parent = HueBg, Color = ColorSequence.new({ColorSequenceKeypoint.new(0, Color3.fromRGB(255,0,0)), ColorSequenceKeypoint.new(0.16, Color3.fromRGB(255,255,0)), ColorSequenceKeypoint.new(0.33, Color3.fromRGB(0,255,0)), ColorSequenceKeypoint.new(0.5, Color3.fromRGB(0,255,255)), ColorSequenceKeypoint.new(0.66, Color3.fromRGB(0,0,255)), ColorSequenceKeypoint.new(0.83, Color3.fromRGB(255,0,255)), ColorSequenceKeypoint.new(1, Color3.fromRGB(255,0,0))}) })
                
                local HuePick = Utility:Create("Frame", { Parent = HueBg, Size = UDim2.new(0, 4, 1, 4), Position = UDim2.new(0, 0, 0, -2), BackgroundColor3 = Color3.fromRGB(255,255,255) })
                
                -- Simulate picking hue
                local sliding = false
                HueBg.InputBegan:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then sliding = true end end)
                UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then sliding = false end end)
                UserInputService.InputChanged:Connect(function(input)
                    if sliding and input.UserInputType == Enum.UserInputType.MouseMovement then
                        local pct = math.clamp((input.Position.X - HueBg.AbsolutePosition.X) / HueBg.AbsoluteSize.X, 0, 1)
                        HuePick.Position = UDim2.new(pct, 0, 0, -2)
                        local color = Color3.fromHSV(pct, 1, 1)
                        Preview.BackgroundColor3 = color
                        if props.Callback then props.Callback(color) end
                    end
                end)
            end

            -- 🟩 COMPONENT: Keybind
            function Elements:CreateKeybind(props)
                local KbFrame = Utility:Create("Frame", { Parent = SecItems, Size = UDim2.new(1, 0, 0, 35), BackgroundColor3 = Obsidian.Themes[Obsidian.CurrentTheme].Element })
                Utility:Create("UICorner", { Parent = KbFrame, CornerRadius = UDim.new(0, 4) })
                Utility:RegisterTheme(KbFrame, "BackgroundColor3", "Element")
                
                local Lbl = Utility:Create("TextLabel", { Parent = KbFrame, Size = UDim2.new(1, -100, 1, 0), Position = UDim2.new(0, 10, 0, 0), BackgroundTransparency = 1, Text = props.Name, Font = Enum.Font.GothamMedium, TextSize = 13, TextColor3 = Obsidian.Themes[Obsidian.CurrentTheme].Text, TextXAlignment = Enum.TextXAlignment.Left })
                Utility:RegisterTheme(Lbl, "TextColor3", "Text")
                
                local Btn = Utility:Create("TextButton", { Parent = KbFrame, Size = UDim2.new(0, 80, 0, 25), Position = UDim2.new(1, -90, 0.5, -12.5), BackgroundColor3 = Obsidian.Themes[Obsidian.CurrentTheme].Stroke, Text = props.Default.Name, Font = Enum.Font.GothamBold, TextSize = 12, TextColor3 = Obsidian.Themes[Obsidian.CurrentTheme].Text, AutoButtonColor = false })
                Utility:Create("UICorner", { Parent = Btn, CornerRadius = UDim.new(0, 4) })
                Utility:RegisterTheme(Btn, "BackgroundColor3", "Stroke")
                
                local key = props.Default
                local mapping = false
                Btn.MouseButton1Click:Connect(function()
                    Btn.Text = "..."
                    mapping = true
                end)
                UserInputService.InputBegan:Connect(function(input)
                    if mapping and input.UserInputType == Enum.UserInputType.Keyboard then
                        key = input.KeyCode
                        Btn.Text = key.Name
                        mapping = false
                    elseif not mapping and input.KeyCode == key then
                        if props.Callback then props.Callback(key) end
                    end
                end)
            end

            return Elements
        end

        return SectionClass
    end

    return WindowClass
end
